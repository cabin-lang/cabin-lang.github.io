foreach number in 1.to(100) {
	if number.mod(3) = 0 and number.mod(5) = 0 {
		terminal.print("FizzBuzz");
	}

	otherwise if number.mod(3) = 0 {
		terminal.print("Fizz");
	}

	otherwise if number.mod(5) = 0 {
		terminal.print("Buzz");
	}

	otherwise {
		terminal.print(number);
	};
}
